ALTER PROCEDURE [dbo].[BD_MONITOREO]
AS
BEGIN
	
	SET NOCOUNT ON;

	--VER DISCOS EN SQL SERVER
		SELECT DISTINCT
			'TAMAÑO DE DISCOS' AS MONITOREO
			,getdate() as FECHA,
			vs.volume_mount_point AS DISCO,
			vs.logical_volume_name AS [NOMBRE LOGICO],
			CONVERT(INT, vs.total_bytes / 1048576.0 /1024) AS [ESPACIO TOTAL (GB)],
			CONVERT(INT, (vs.total_bytes - vs.available_bytes) / 1048576.0 / 1024) AS [ESPACIO UTILIZADO (GB)],
			CONVERT(INT, vs.available_bytes / 1048576.0 /1024) AS [ESPACIO DISPONIBLE (GB)],
			FORMAT((vs.available_bytes * 100.0 / vs.total_bytes), 'N2') AS [PORCENTAJE DISPONIBLE],
			CASE 
				WHEN (vs.available_bytes * 100.0 / vs.total_bytes) < 10 THEN 'Crítico'
				ELSE 'Estable'
			END AS COMPORTAMIENTO
		FROM 
			sys.master_files AS mf
		CROSS APPLY 
			sys.dm_os_volume_stats(mf.database_id, mf.file_id) AS vs
		ORDER BY 
			DISCO ASC;

    -- ESPERAS
	 SET  NOCOUNT ON
	;with monitor as (
    SELECT
	   [ID ESPERA]        = er.session_id 
      ,[DASE DE DATOS]            = DB_Name(er.database_id)
      ,[TIPO DE COMANDO]    = er.command  
	  ,[SEGUNDOS TRANSCURRIDOS]        = er.total_elapsed_time/1000 
      ,[SQL]    = st.text
      ,[PROCESO EJECUTANDOSE]=
                        CASE WHEN er.[statement_start_offset] > 0 THEN 
                            CASE er.[statement_end_offset]     WHEN -1 THEN SUBSTRING(st.TEXT, (er.[statement_start_offset]/2) + 1, 2147483647)
                                ELSE SUBSTRING(st.TEXT, (er.[statement_start_offset]/2) + 1, (er.[statement_end_offset] - er.[statement_start_offset])/2)  
                            END 
                        ELSE 
                            CASE er.[statement_end_offset]     WHEN -1 THEN RTRIM(LTRIM(st.[text])) 
                                ELSE LEFT(st.TEXT, (er.[statement_end_offset]/2) +1) 
                            END 
                        END 
     
      ,[TIEMPO CPU]        = er.cpu_time 
     -- ,[LECTURAS DE ENTRADA Y SALIDA]        = er.logical_reads + er.reads
     -- ,[ESCRITURA DE ENTRAD Y SALIDA]        = er.writes 
      ,[HORA INICIO]        = er.start_time   
      
    FROM SYS.DM_EXEC_REQUESTS er    
    OUTER APPLY SYS.DM_EXEC_SQL_TEXT(er.sql_handle) st    
    LEFT JOIN SYS.DM_EXEC_SESSIONS ses  ON ses.session_id = er.session_id    
    LEFT JOIN SYS.DM_EXEC_CONNECTIONS con ON con.session_id = ses.session_id  
	OUTER APPLY sys.dm_exec_query_plan (er.plan_handle) eqp
    WHERE er.session_id!=@@SPID and st.text IS NOT NULL)
   

	select 'ESPERAS' as MONITOREO, getdate() as FECHA, * from monitor 
	where  [SEGUNDOS TRANSCURRIDOS] > 5 AND
	[DASE DE DATOS] != 'master'
	order by [SEGUNDOS TRANSCURRIDOS] desc



	--VER BLOQUEOS CON KILL INCLUIDO	  
    SELECT  [MONITOREO] = 'BLOQUEOS'
		,getdate() as FECHA
		,session_id AS [ID BLOQUEO]
        ,[MATAR PROCESO] = 'kill '+ convert(varchar(20),blocking_session_id)
        ,wait_time AS [TIEMPO DE ESPERA]
        ,wait_type AS [TIPO DE ESPERA]
       -- ,last_wait_type
       -- ,wait_resource
       -- ,transaction_isolation_level
        ,lock_timeout AS [TIEMPO DEL BLOQUEO]
    FROM sys.dm_exec_requests
    WHERE blocking_session_id <> 0
    
	-- VALIDA SI EXISTE BLOQUEOS LO ALMACENA EN LAS TABLAS DE ALERTASDB.dbo.BI_BEL_BLOQUEOS
	IF EXISTS(Select top 1 1 FROM sys.dm_exec_requests WHERE blocking_session_id <> 0)
	BEGIN
		 Exec BD_CAPTURA_BLOQUEOS
		 PRINT 'Bloqueo almacenado exitosamente en ALERTASDB.dbo.BI_BEL_BLOQUEOS'
	END


	/*Total de conexiones*/
	Declare @ParametroFullConexiones DECIMAL = 32767
	
	select  [MONITOREO] = 'CONEXIONES POR CLUSTER', getdate() as FECHA, count(*) as [TOTAL DE CONEXIONES]
	, cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2))  as [PORCENTAJE]
	 ,CASE 
			when cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) <= 60 THEN 'NORMAL'
			when cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) > 90 THEN 'EXCESO DE CONEXIONES - REVISAR'
			when (cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) > 60 AND (CAST(Count(*) AS decimal)/CAST(32767 as decimal)) <= 90 ) THEN 'CARGA DE CONEXIONES'			
		END AS COMPORTAMIENTO
	from sys.sysprocesses
	WHERE 
   		dbid > 0


	/*TOTAL DE CONEXIONS POR BD*/
	SET @ParametroFullConexiones = 30000
	SELECT TOP 5 [MONITOREO] = 'CONEXIONES POR BD', getdate() AS FECHA,
   		DB_NAME(dbid) AS [BASE DE DATOS]
        , COUNT(dbid) AS CONEXIONES
		, cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2))  as PORCENTAJE
	    ,CASE 
			when cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) <= 60 THEN 'NORMAL'
			when cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) > 90 THEN 'EXCESO DE CONEXIONES - REVISAR'
			when (cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) > 60 AND (CAST(Count(*) AS decimal)/CAST(32767 as decimal)) <= 90 ) THEN 'CARGA DE CONEXIONES'			
		END AS COMPORTAMIENTO
	FROM sys.sysprocesses WHERE dbid > 0 
	GROUP BY dbid
	order by COUNT(dbid) desc;


	DECLARE @BD VARCHAR(150) = (
 	  SELECT TOP 1  DB_NAME(dbid) AS BaseDatos
				FROM sys.sysprocesses WHERE dbid > 0 GROUP BY 
   				dbid, loginame     order by COUNT(dbid) desc)
	

	/*CONEXIONES POR USUARIO*/
	SET @ParametroFullConexiones = 20000
	SELECT TOP 5  [MONITOREO] = 'CONEXIONES POR USUARIO', getdate() AS FECHA,
   		DB_NAME(dbid) AS [BASE DE DATOS], COUNT(dbid) AS CONEXIONES, loginame AS [USUARIO]	
		,CASE 
			when cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) <= 60 THEN 'NORMAL'
			when cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) > 90 THEN 'EXCESO DE CONEXIONES - REVISAR'
			when (cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) > 60 AND (CAST(Count(*) AS decimal)/CAST(32767 as decimal)) <= 90 ) THEN 'CARGA DE CONEXIONES'			
		END AS COMPORTAMIENTO
	FROM sys.sysprocesses WHERE dbid > 0 GROUP BY 
   		dbid, loginame     order by COUNT(dbid) desc;


	/*Vista por BD*/
	SET @ParametroFullConexiones = 1000
	
	SELECT TOP 5 [MONITOREO] = 'CONEXIONES POR HOST',  getdate() AS FECHA,
	hostname AS [NOMBRE DEL HOST], count(*) as CONEXIONES, DB_NAME(dbid) AS [BASE DE DATOS]
	,CASE 
			when cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) <= 60 THEN 'NORMAL'
			when cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) > 90 THEN 'EXCESO DE CONEXIONES - REVISAR'
			when (cast( (CAST(Count(*) AS decimal)/CAST(@ParametroFullConexiones as decimal)) *100 as decimal (20,2)) > 60 AND (CAST(Count(*) AS decimal)/CAST(32767 as decimal)) <= 90 ) THEN 'CARGA DE CONEXIONES'			
		END AS COMPORTAMIENTO
	FROM sys.sysprocesses WHERE dbid > 0  AND DB_NAME(dbid) = @BD
	group by hostname, DB_NAME(dbid)  
	order by conexiones desc


	--LOGSPACE
	CREATE TABLE #LogSpaceResults (
		DatabaseName sysname,
		LogSizeMB float,
		LogSpaceUsedPercent float,
		Status int
	);
	INSERT INTO #LogSpaceResults (DatabaseName, LogSizeMB, LogSpaceUsedPercent, Status)
	EXEC('DBCC SQLPERF(LOGSPACE)');

	select [MONITOREO] = 'ESPACIO DEL LOG',  getdate() AS FECHA, 
		DatabaseName AS [BASE DE DATOS]
		,CAST(LogSizeMB AS  decimal(18,2)) AS [TAMAÑO LOG (MB)]
		,CAST( CAST( LogSpaceUsedPercent AS decimal(18,2)) AS varchar) + ' %' AS [ESPACIO UTILIZADO LOG]
		,d.state_desc as ESTATUS
		,d.recovery_model_desc as [MODO RECUPERACION]
		,CASE WHEN  CAST( LogSpaceUsedPercent AS decimal(18,2)) > 90 THEN '% LOG ALTO'
			 WHEN  CAST( LogSpaceUsedPercent AS decimal(18,2)) BETWEEN 80 AND 90 THEN 'REVISAR'
		ELSE 'NORMAL'
		End as COMPORTAMIENTO
	from #LogSpaceResults l
	INNER JOIN Sys.databases d on d.name = l.DatabaseName

	Drop table #LogSpaceResults


	--REVISIÓN DE JOB'S EJECUTANDOCE
	SELECT [MONITOREO] = 'ESTADO JOBs',  getdate() AS FECHA, 
		sj.Name as [NOMBRE JOB],
			CASE
				when sj.enabled = 1 then 'Enabled'
				else 'Disabled'
			end as [ESTADO],
			CASE
				WHEN sja.start_execution_date IS NULL THEN 'Not running'
				WHEN sja.start_execution_date IS NOT NULL AND sja.stop_execution_date IS NULL THEN 'Running'
				WHEN sja.start_execution_date IS NOT NULL AND sja.stop_execution_date IS NOT NULL THEN 'Not running'
			END AS 'ESTADO EJECUCIÓN',
			cast(sjs.next_run_date as varchar(8)) AS [FECHA SIGUIENTE EJECUCIÓN],
			RIGHT('0' + CAST(sjs.next_run_time / 10000 AS VARCHAR(2)), 2) + ':' +
			RIGHT('0' + CAST((sjs.next_run_time % 10000) / 100 AS VARCHAR(2)), 2) + ':' +
			RIGHT('0' + CAST(sjs.next_run_time % 100 AS VARCHAR(2)), 2) AS [HORA SIGUIENTE EJECUCIÓN]
		FROM msdb.dbo.sysjobs sj
		JOIN msdb.dbo.sysjobactivity sja ON sj.job_id = sja.job_id
		join msdb.dbo.sysjobschedules sjs ON sj.job_id=sjs.job_id
		WHERE session_id = (
			SELECT MAX(session_id) FROM msdb.dbo.sysjobactivity)
		ORDER BY 'ESTADO EJECUCIÓN' desc, sj.enabled DESC

	--REVISION DE BACKUP EJECUTANDOCE	
	SELECT  [MONITOREO] = 'BACKUP EN EJECUCION', GETDATE() as FECHA, SESSION_ID [ID SESION] , '[' + CAST( DATABASE_ID AS VARCHAR(10 )) + '] ' + DB_NAME (DATABASE_ID) AS [DASE DE BATOS]
		, PERCENT_COMPLETE [%] ,        START_TIME [TIEMPO DE INICIO] 
		, STATUS [ESTATUS], COMMAND [COMANDO] ,      DATEADD(MS , ESTIMATED_COMPLETION_TIME, GETDATE()) AS [TIEMPO DE ESPERA] 
		, CPU_TIME [TIEMPO DE CPU]
		FROM SYS .DM_EXEC_REQUESTS    
		WHERE        COMMAND LIKE '%BACKUP%'
		OR COMMAND LIKE '%RESTORE%'
		OR COMMAND LIKE '%DBCC%'

	-- MIRRORING 
	SELECT GETDATE() as FECHA,
			ag.name AS AvailabilityGroupName,
			dt.name AS DatabaseName,   
			ags.primary_replica AS PrimaryReplica,
				ar.replica_server_name AS ReplicaServerName,
			--agd.role_desc AS ReplicaRole,
			agd.synchronization_state_desc AS SyncState,
			ags.synchronization_health_desc AS SyncHealth,
			case when agd.synchronization_state_desc = 'NOT SYNCHRONIZING' THEN 'NO SINCRONIZADO'
				 Else 'OK'
			End as Estatus
		FROM 
			sys.availability_groups AS ag
		INNER JOIN 
			sys.dm_hadr_availability_group_states AS ags 
			ON ag.group_id = ags.group_id
		INNER JOIN 
			sys.availability_replicas AS ar 
			ON ag.group_id = ar.group_id
		INNER JOIN 
			sys.dm_hadr_database_replica_states AS agd
			ON ags.group_id = agd.group_id
		INNER JOIN sys.databases dt
			on agd.database_id = dt.database_id
		WHERE agd.is_local = 1
		ORDER BY 
			agd.database_id, ag.name, ar.replica_server_name;
END
GO
